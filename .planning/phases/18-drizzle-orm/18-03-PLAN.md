<!-- Drizzle upsert pattern (from research) -->

```typescript
import { db } from "@/lib/db";
import { aptTransactions } from "@/lib/db/schema";
import { sql } from "drizzle-orm";

// Upsert (insert or update on conflict)
await db.insert(aptTransactions)
  .values(rows)
  .onConflictDoUpdate({
    target: aptTransactions.id,
    set: {
      tradePrice: sql`excluded.trade_price`,
      isNewHigh: sql`excluded.is_new_high`,
    },
  });

// Insert ignore duplicates
await db.insert(aptRentTransactions)
  .values(rows)
  .onConflictDoNothing();

// Delete
await db.delete(table).where(eq(table.id, id));

// Update
await db.update(table).set({ field: value }).where(eq(table.id, id));
```

    For each file:
    1. Replace imports: `createServiceClient` → `import { db } from "@/lib/db"` + schema imports
    2. Convert SELECT queries: `.from("table").select(...).eq(...)...` → `db.select({...}).from(table).where(eq(...))`
    3. Convert INSERT queries: `.from("table").insert(data)` → `db.insert(table).values(data)`
    4. Convert UPDATE queries: `.from("table").update(data).eq(...)` → `db.update(table).set(data).where(eq(...))`
    5. Convert UPSERT queries: `.from("table").upsert(data, { onConflict: "id" })` → `db.insert(table).values(data).onConflictDoUpdate/DoNothing()`
    6. Remove `{ data, error }` destructuring — Drizzle returns results directly

    Special cases:
    - `pageview/route.ts`: Has INSERT for page_views — use `db.insert(pageViews).values({...})`
    - `push/subscribe/route.ts`: Has INSERT for push_subscriptions — use `db.insert(pushSubscriptions).values({...})`
    - `dam/content/route.ts`: May have updates to content_queue status

    Column name casing: API routes return JSON to clients. If the API response uses snake_case (matching current behavior), use snake_case select aliases: `.select({ trade_price: aptTransactions.tradePrice })`. Check each file to determine if the response body is consumed by frontend code expecting snake_case.

    Run `npx tsc --noEmit` after completing all 14 files.
