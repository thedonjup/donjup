```typescript
// inArray for dynamic IN clauses
import { db } from "@/lib/db";
import { aptTransactions } from "@/lib/db/schema";
import { and, eq, inArray, asc, sql } from "drizzle-orm";

// Guard empty arrays (Drizzle inArray([]) can fail)
if (regionCodes.length === 0) return [];

const rows = await db
  .select({
    trade_date: aptTransactions.tradeDate,
    trade_price: aptTransactions.tradePrice,
    floor: aptTransactions.floor,
    deal_type: aptTransactions.dealType,
  })
  .from(aptTransactions)
  .where(and(
    inArray(aptTransactions.regionCode, regionCodes),
    eq(aptTransactions.propertyType, 1)
  ))
  .orderBy(asc(aptTransactions.tradeDate));

// Dynamic filter array for search
const filters: SQL[] = [];
if (q) filters.push(ilike(aptComplexes.aptName, `%${q}%`));
if (minPrice) filters.push(gte(aptTransactions.tradePrice, minPrice));
const results = await db.select(...)
  .from(aptComplexes)
  .where(filters.length > 0 ? and(...filters) : undefined);

// Raw SQL via Drizzle (for DDL or very complex queries)
import { sql } from "drizzle-orm";
await db.execute(sql`CREATE TABLE IF NOT EXISTS analytics_daily (...)`);
```
