/** 층별 고층 환산 계수 (저층 할인율의 역수) */
export const FLOOR_ADJUSTMENT_FACTORS: Record<number, number> = {
  1: 1.1494,  // 1층 -13% → ×(1/0.87)
  2: 1.1111,  // 2층 -10% → ×(1/0.90)
  3: 1.0417,  // 3~4층 -4% → ×(1/0.96)
};

2. Add adjustFloorPrice function (after FLOOR_ADJUSTMENT_FACTORS):

```typescript
/**
 * 저층 거래가를 고층 기준가로 환산 (NORM-02)
 *
 * 1층: ×1.1494, 2층: ×1.1111, 3층: ×1.0417
 * 4층 이상은 환산 없이 원가 반환
 */
export function adjustFloorPrice(tradePrice: number, floor: number): number {
  const factor = FLOOR_ADJUSTMENT_FACTORS[floor];
  if (!factor) return tradePrice;
  return Math.round(tradePrice * factor);
}
```

3. Modify filterTransactions signature — change `excludeLowFloor: boolean` to `lowFloorMode: 'adjust' | 'include' | 'exclude'`:

```typescript
export function filterTransactions<
  T extends { floor: number; trade_price: number; deal_type: string | null }
>(
  txns: T[],
  opts: { lowFloorMode: 'adjust' | 'include' | 'exclude'; recentMedian: number }
): { normal: (T & { adjusted_price?: number })[]; directDeals: T[]; excluded: T[] }
```

- `'adjust'` (default behavior, toggle OFF): Low-floor transactions get adjusted price and are included in normal. Add `adjusted_price` field to the returned object, and set `trade_price` on a shallow copy to the adjusted value so downstream consumers (groupByMonth, trendLine) use the adjusted price.
- `'include'` (toggle ON): Low-floor transactions included in normal at original price (no adjustment).
- `'exclude'`: Low-floor transactions go to excluded array (backward compat, not currently used but keeps API clean).

Implementation change in the for loop:

```typescript
const isLowFloor = t.floor <= LOW_FLOOR_MAX;

if (isLowFloor) {
  if (lowFloorMode === 'exclude') {
    excluded.push(t);
    continue;
  }
  if (lowFloorMode === 'adjust') {
    const adjustedPrice = adjustFloorPrice(t.trade_price, t.floor);
    // Create shallow copy with adjusted trade_price
    const adjusted = { ...t, trade_price: adjustedPrice, original_price: t.trade_price };
    // Continue to median/direct-deal checks below with adjusted price
    // (use adjustedPrice for median comparison)
    if (recentMedian > 0 && adjustedPrice < recentMedian * 0.90) {
      excluded.push(t);
      continue;
    }
    if (isDirectDeal(t.deal_type)) {
      directDeals.push(adjusted as T);
      continue;
    }
    normal.push(adjusted as T);
    continue;
  }
  // lowFloorMode === 'include': fall through to normal processing with original price
}
```

The rest of the function (median check, direct deal check, normal push) stays the same for non-low-floor transactions.
