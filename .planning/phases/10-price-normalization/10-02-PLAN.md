interface ChartTransaction {
  trade_date: string;
  trade_price: number;
  size_sqm: number;
  deal_type: string | null;
  floor: number;
  isDirectDeal: boolean;  // pre-computed by parent
  prevPrice?: number;     // for direct deals: previous same-size non-direct transaction price
}

interface TrendPoint {
  month: string;       // YYYY-MM
  median: number;      // 3-month moving median price
  count: number;       // transaction count in window
  isLowConfidence: boolean;  // < 5 transactions
}

interface PriceHistoryChartProps {
  normalDots: ChartTransaction[];      // normal transactions (green dots)
  directDealDots: ChartTransaction[];  // direct deals (gray transparent dots with prevPrice connectors)
  trendLine: TrendPoint[];             // 3-month moving median line
  sizeUnit?: "sqm" | "pyeong";
}
```

   const recentMedian = useMemo(() => {

     const dateStr = last6Months.toISOString().slice(0, 10);
     const recentPrices = saleTxns
       .filter(t => t.size_sqm === selectedSize && t.trade_date >= dateStr && t.floor > LOW_FLOOR_MAX)
       .map(t => t.trade_price);
     return computeMedianPrice(recentPrices);
   }, [saleTxns, selectedSize]);
   ```

   const { normal, directDeals } = useMemo(() => {

     return filterTransactions(sizeFiltered, {
       excludeLowFloor: !includeLowFloor,
       recentMedian,
     });
   }, [saleTxns, selectedSize, includeLowFloor, recentMedian]);
   ```

5. **Compute trend line data:**

   const trendLine = useMemo(() => {
     const monthly = groupByMonth(normal);
     return computeMovingMedian(monthly);
   }, [normal]);
   ```

   ```typescript
   const directDealsWithPrev = useMemo(() => {
     // Sort all same-size non-direct transactions by date descending for lookup
     const normalByDate = [...normal].sort((a, b) => b.trade_date.localeCompare(a.trade_date));

     return directDeals.map(dd => {
       // Find the most recent non-direct transaction before this direct deal's date
       const prev = normalByDate.find(n => n.trade_date <= dd.trade_date);
       return {
         trade_date: dd.trade_date,
         trade_price: dd.trade_price,
         size_sqm: dd.size_sqm,
         deal_type: dd.deal_type,
         floor: dd.floor,
         isDirectDeal: true,
         prevPrice: prev?.trade_price,
       };
     });
   }, [directDeals, normal]);
   ```
